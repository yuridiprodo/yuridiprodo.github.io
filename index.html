async function loadArticles() { const response = await fetch('articoli/'); // Cambia il percorso se necessario const text = await response.text(); const parser = new DOMParser(); const htmlDoc = parser.parseFromString(text, 'text/html'); const links = Array.from(htmlDoc.querySelectorAll('a')); const articles = []; for (const link of links) { const fileName = link.getAttribute('href'); if (fileName.endsWith('.md')) { const fileResponse = await fetch(`articoli/${fileName}`); const fileText = await fileResponse.text(); const titleMatch = fileText.match(/title:\s*"(.*?)"/); const dateMatch = fileText.match(/date:\s*(\d{4}-\d{2}-\d{2})/); if (titleMatch && dateMatch) { articles.push({ title: titleMatch[1], file: fileName, date: new Date(dateMatch[1]) }); } } } // Ordina gli articoli per data, dal piÃ¹ recente articles.sort((a, b) => b.date - a.date); const articleList = document.getElementById('article-list'); articles.forEach(article => { const li = document.createElement('li'); li.innerHTML = `${article.title}`; articleList.appendChild(li); }); } loadArticles();
